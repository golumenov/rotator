Implementation

// Define settings
var settings={position: 'left'};
// Define click event
var customEvent=function(currentIndex){
    alert('Show image #'+currentIndex);
}
// Use plugin
$('#ag_rotator').ag_rotator(settings,customEvent)
        // Plugin supports the call chain
        .find('img').each(function(element,index){
        $(this).css('border','1px solid #ffffff');
});
